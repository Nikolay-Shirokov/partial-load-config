# Пример файла со списком объектов для частичной выгрузки конфигурации
#
# Формат: по одному объекту на строку
# Имена объектов указываются в формате 1С: Тип.Имя
#
# Поддерживаемые форматы:
# - Объекты метаданных: Справочник.Номенклатура
# - Документы: Документ.РеализацияТоваров
# - Отчеты: Отчет.АнализПродаж
# - Обработки: Обработка.ЗагрузкаДанных
# - Свойства конфигурации: Configuration.Help
#
# Строки, начинающиеся с #, игнорируются
# Пустые строки игнорируются

# Справочники
Справочник.Номенклатура
Справочник.Контрагенты

# Документы
Документ.РеализацияТоваров
Документ.ПоступлениеТоваров

# Отчеты
Отчет.АнализПродаж

# Обработки
Обработка.ЗагрузкаДанных

# Справка конфигурации
Configuration.Help

# Заставка конфигурации
Configuration.Splash
